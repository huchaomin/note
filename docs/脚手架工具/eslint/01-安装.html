# 最终生成的配置文件.eslintrc.js

module.exports = {
  'env': {
    'browser': true,
    'es2021': true,
  },
  'extends': [
    'plugin:vue/essential',
    'standard',
  ],
  'parserOptions': {
    'ecmaVersion': 6, // 这里不知道为啥生成13了，我们手动改成6 
    'sourceType': 'module',
  },
  'plugins': [
    'vue',
  ],
  'rules': {},
}
